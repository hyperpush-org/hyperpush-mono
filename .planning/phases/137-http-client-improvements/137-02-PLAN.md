**Function 4: mesh_http_stream(req_handle: u64, callback_fn: *mut u8, callback_env: *mut u8) -> *mut u8**
OS-thread-per-stream model. Returns Result<HttpResponse, String> with empty body string:
```rust
#[no_mangle]
pub extern "C" fn mesh_http_stream(
    req_handle: u64,
    callback_fn: *mut u8,
    callback_env: *mut u8,
) -> *mut u8 {
    unsafe {
        let req_data = &*(req_handle as *const MeshRequestData);

        // Build and execute request synchronously to get status/headers first
        // (before consuming the body into the streaming thread)
        let agent = build_one_shot_agent(req_data.timeout_ms);
        let request = build_request(&agent, req_data);
        match request.http_status_as_error(false).call() {
            Err(e) => {
                // Network error before even starting stream
                let msg = format_error(&e);
                let msg_mesh = mesh_string_new(msg.as_ptr(), msg.len() as u64);
                return alloc_result(1, msg_mesh as *mut u8) as *mut u8;
            }
            Ok(response) => {
                let status = response.status() as i64;
                let headers_map = build_headers_map_from_response(&response);

                // Consume body into OS thread
                let cancel = Arc::new(AtomicBool::new(false));
                let cancel_clone = cancel.clone();
                let mut body = response.into_body();

                let fn_ptr = callback_fn;
                let env_ptr = callback_env;

                std::thread::spawn(move || {
                    let mut reader = body.as_reader();
                    let mut buf = vec![0u8; 8192];
                    loop {
                        if cancel_clone.load(Ordering::SeqCst) { break; }
                        match reader.read(&mut buf) {
                            Ok(0) => break,
                            Ok(n) => {
                                let chunk = mesh_string_new(buf.as_ptr(), n as u64);
                                // Dispatch callback: fn(chunk) or fn(env, chunk)
                                let result = if env_ptr.is_null() {
                                    let f: fn(*mut u8) -> *mut u8 = std::mem::transmute(fn_ptr);
                                    f(chunk as *mut u8)
                                } else {
                                    let f: fn(*mut u8, *mut u8) -> *mut u8 = std::mem::transmute(fn_ptr);
                                    f(env_ptr, chunk as *mut u8)
                                };
                                if is_stop_atom(result) { break; }
                            }
                            Err(_) => break,
                        }
                    }
                    // cancel handle dropped here (Arc ref count decremented)
                    let _ = cancel_clone;
                });

                // Return Ok(HttpResponse) with status + headers + empty body
                let empty_body = mesh_string_new(b"".as_ptr(), 0);
                alloc_client_response_result(status, empty_body as *mut u8, headers_map) as *mut u8
            }
        }
    }
}
```

// Http.stream(req, fn(String) -> Atom) -> Result<HttpResponse, String>
// Callback type: fn(String) -> Atom  OR  fn(Env, String) -> Atom (closure)
// At builtins level we register the simplest (non-closure) variant — typeck handles dispatch
let stream_cb_t = Ty::fun(vec![Ty::string()], Ty::Con(TyCon::new("Atom")));  // or Ty::string() if Atom not defined
// Actually use a fresh type variable for callback since typeck cannot express fn closures in builtins easily.
// Register as taking a Ptr for callback to allow both bare fn and closure:
env.insert("http_stream".into(), Scheme::mono(Ty::fun(
    vec![http_req_t.clone(), Ty::fun(vec![Ty::string()], Ty::Con(TyCon::new("String")))],
    Ty::result(http_resp_t.clone(), Ty::string())
)));
env.insert("http_stream_bytes".into(), Scheme::mono(Ty::fun(
    vec![http_req_t.clone(), Ty::fun(vec![Ty::string()], Ty::Con(TyCon::new("String")))],
    Ty::result(http_resp_t.clone(), Ty::string())
)));
// Http.client() -> Client (Int handle, same opaque pattern as Request)
let http_client_t = Ty::int(); // Agent handle as u64
env.insert("http_client".into(), Scheme::mono(Ty::fun(vec![], http_client_t.clone())));
// Http.send_with(client, req) -> Result<HttpResponse, String>
env.insert("http_send_with".into(), Scheme::mono(Ty::fun(
    vec![http_client_t.clone(), http_req_t.clone()],
    Ty::result(http_resp_t.clone(), Ty::string())
)));
// Http.client_close(client) -> ()
env.insert("http_client_close".into(), Scheme::mono(Ty::fun(
    vec![http_client_t.clone()],
    Ty::Tuple(vec![])
)));
